### NakamaClient.gd (autoload pattern)
```gdscript
extends Node

# Singleton wrapper around the Nakama SDK client.
# Reads host/port/scheme from project settings (set via export var override).

const NAKAMA_HOST_DEFAULT := "REPLACE_WITH_RAILWAY_URL_HOST_ONLY"  # ej: "barrabrava-nakama-production.up.railway.app"
const NAKAMA_PORT_DEFAULT := 443
const NAKAMA_SCHEME_DEFAULT := "https"
const NAKAMA_SERVER_KEY_DEFAULT := "defaultkey"  # OVERRIDE via project settings or env at runtime

@export var nakama_host: String = NAKAMA_HOST_DEFAULT
@export var nakama_port: int = NAKAMA_PORT_DEFAULT
@export var nakama_scheme: String = NAKAMA_SCHEME_DEFAULT
@export var nakama_server_key: String = NAKAMA_SERVER_KEY_DEFAULT

var client  # NakamaClient instance (from SDK)

func _ready() -> void:
    # Lazy-load the SDK to avoid load order issues
    var NakamaSDK = preload("res://addons/com.heroiclabs.nakama/Nakama.gd")
    client = NakamaSDK.create_client(
        nakama_server_key,
        nakama_host,
        nakama_port,
        nakama_scheme,
        10  # timeout seconds
    )
    print("[NakamaClient] initialized: %s://%s:%d" % [nakama_scheme, nakama_host, nakama_port])
```

### AuthManager.gd (RESEARCH §1 + §6 verbatim adaptado)
```gdscript
extends Node

# Manages Nakama session lifecycle: persist, restore, refresh.
# Session is stored in user://session.cfg via ConfigFile (Phase 1 — not Keychain).

const SESSION_FILE := "user://session.cfg"

signal session_ready(session)
signal session_cleared

var session  # NakamaSession instance or null

func _ready() -> void:
    await get_tree().process_frame  # wait for NakamaClient autoload to init
    await _try_restore_session()

func login(email: String, password: String) -> Dictionary:
    var result = await NakamaClient.client.authenticate_email_async(email, password, false, "")
    if result.is_exception():
        return {"ok": false, "error": str(result.get_exception().message)}
    session = result
    _save_session()
    session_ready.emit(session)
    return {"ok": true}

func register(email: String, password: String) -> Dictionary:
    var result = await NakamaClient.client.authenticate_email_async(email, password, true, "")
    if result.is_exception():
        return {"ok": false, "error": str(result.get_exception().message)}
    session = result
    _save_session()
    session_ready.emit(session)
    return {"ok": true}

func logout() -> void:
    session = null
    var fa = FileAccess.open(SESSION_FILE, FileAccess.WRITE)
    if fa:
        fa.close()
    DirAccess.remove_absolute(ProjectSettings.globalize_path(SESSION_FILE))
    session_cleared.emit()

func is_authenticated() -> bool:
    return session != null and not session.expired

func _save_session() -> void:
    var cfg = ConfigFile.new()
    cfg.set_value("auth", "token", session.token)
    cfg.set_value("auth", "refresh_token", session.refresh_token)
    cfg.save(SESSION_FILE)
    print("[AuthManager] session saved")

func _try_restore_session() -> void:
    var cfg = ConfigFile.new()
    if cfg.load(SESSION_FILE) != OK:
        print("[AuthManager] no saved session")
        return
    var token = cfg.get_value("auth", "token", "")
    var refresh = cfg.get_value("auth", "refresh_token", "")
    if token == "":
        return
    var restored = NakamaClient.client.restore_session(token)
    if restored.expired:
        if refresh == "":
            print("[AuthManager] session expired, no refresh token")
            return
        var refreshed = await NakamaClient.client.session_refresh_async(restored, refresh)
        if refreshed.is_exception():
            print("[AuthManager] refresh failed: %s" % refreshed.get_exception().message)
            return
        session = refreshed
    else:
        session = restored
    _save_session()
    session_ready.emit(session)
    print("[AuthManager] session restored, user_id=%s" % session.user_id)
```
**ASSUMED A1 de RESEARCH:** field name es `session.token` — si en Plan 03 fallan auth tests, swap a `session.auth_token`. Documentar la decisión.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Godot client → Nakama server | Cliente envía credenciales auth; Nakama responde con session token |
| user:// ConfigFile → device storage | Token JWT persiste sin encriptar en sandbox de app |
| Godot addon → SDK code | Código de terceros (Nakama SDK) ejecuta dentro del cliente |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-1-CLI-01 | Information Disclosure | session.cfg sin encriptar | accept | Phase 1 alcance: user:// está sandboxed por OS (Android: /data/data/{app}/files no accesible sin root; iOS: app Documents en sandbox). Encriptación verdadera (Keychain/Keystore) requiere plugin nativo — deferred per RESEARCH §1 Token persistence note. |
| T-1-CLI-02 | Spoofing | NakamaClient hardcoded server key | mitigate | Server key en código es la SAME que Railway env var (Nakama valida match). Para Phase 1, hardcoded en `NakamaClient.gd` es aceptable (RESEARCH §1). Para Phase 7 release: extraer a build config o flag de export. |
| T-1-CLI-03 | Tampering | Nakama SDK addon — supply chain | mitigate | SDK descargado de github.com/heroiclabs (oficial), versión pinneada v3.4.0 (NO main branch). Verificar checksum/release notes contra repo oficial antes de commitear. |
| T-1-CLI-04 | Denial of Service | Auto-restore session loop infinito si refresh siempre falla | mitigate | `AuthManager._try_restore_session()` solo intenta una vez al boot; si refresh falla, deja `session = null` y log warning. No hay retry automático. |
| T-1-CLI-05 | Information Disclosure | Logs del cliente leakean session.token | mitigate | Verificar que ningún `print()` en AuthManager / NakamaClient imprime `session.token` o `session.refresh_token` en plano. Inspeccionar grep: `grep -E "print.*token" scripts/autoloads/*.gd` — debe estar vacío excepto el log "session restored, user_id=..." que NO incluye el token. |
</threat_model>
