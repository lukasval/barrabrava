```gdscript
# FCMPlugin is the Android FCM plugin installed in plan 02-07.
# Connect on_token_received signal so we capture the token when it arrives.
# Note: Engine.get_singleton returns null if not on Android or plugin not loaded.
func _ready() -> void:
    # ... existing client setup code stays here ...
    
    # Phase 2: wire FCM token signal if plugin available.
    _wire_fcm_plugin()

func _wire_fcm_plugin() -> void:
    var fcm = Engine.get_singleton("FCMPlugin") if Engine.has_singleton("FCMPlugin") else null
    if fcm == null:
        push_warning("[NakamaService] FCMPlugin not available (non-Android build or plugin missing)")
        return
    if not fcm.on_token_received.is_connected(_on_fcm_token_received):
        fcm.on_token_received.connect(_on_fcm_token_received)
    # Force token emission — some implementations only emit on refresh, not initial fetch.
    fcm.getToken()

func _on_fcm_token_received(token: String) -> void:
    if token.is_empty():
        push_warning("[NakamaService] FCM token received but empty — skipping registration")
        return
    # Register on server (fire-and-forget — no await here to avoid blocking signal handler).
    register_fcm_token.call_deferred(token, "android")
```

Now add the three async methods at the bottom of the file:

```gdscript
# Phase 2: FCM token registration.
# Called via _on_fcm_token_received signal handler (deferred).
func register_fcm_token(token: String, platform: String) -> Dictionary:
    if not AuthManager.is_authenticated():
        return {"ok": false, "error": "not_authenticated"}
    var session = AuthManager.session
    var payload = JSON.stringify({"token": token, "platform": platform})
    var resp = await client.rpc_async(session, "register_fcm_token", payload)
    if resp.is_exception():
        push_warning("[NakamaService] register_fcm_token failed: " + str(resp.get_exception().message))
        return {"ok": false, "error": str(resp.get_exception().message)}
    var data = JSON.parse_string(resp.payload)
    if typeof(data) != TYPE_DICTIONARY:
        return {"ok": false, "error": "invalid_response"}
    return {"ok": true, "data": data}

# Phase 2: Subscribe device to FCM club topic (after club pick + on app resume).
# Idempotent: if already subscribed, skip and return {ok:true, already_subscribed:true}.
# D-09: Club topic = "club_" + club_id. Player subscribes post-club-pick.
func subscribe_to_club_topic(club_id: String) -> Dictionary:
    var topic = AppConfig.FCM_TOPIC_PREFIX + club_id
    if PlayerStore.subscribed_topics.has(topic):
        return {"ok": true, "already_subscribed": true}
    var fcm = Engine.get_singleton("FCMPlugin") if Engine.has_singleton("FCMPlugin") else null
    if fcm == null:
        push_warning("[NakamaService] FCMPlugin not available — cannot subscribe to topic " + topic)
        return {"ok": false, "error": "fcm_unavailable"}
    fcm.subscribeToTopic(topic)
    PlayerStore.subscribed_topics.append(topic)
    return {"ok": true, "topic": topic}

# Phase 2: Current/upcoming match window for this player's club.
# HomeScreen calls this on _ready() and _notification(NOTIFICATION_APPLICATION_RESUMED).
func get_current_window() -> Dictionary:
    if not AuthManager.is_authenticated():
        return {"ok": false, "error": "not_authenticated"}
    var session = AuthManager.session
    var resp = await client.rpc_async(session, "get_current_window", "{}")
    if resp.is_exception():
        return {"ok": false, "error": str(resp.get_exception().message)}
    var data = JSON.parse_string(resp.payload)
    if typeof(data) != TYPE_DICTIONARY:
        return {"ok": false, "error": "invalid_response"}
    return {"ok": true, "data": data}
```

Note: AuthManager and NakamaService are both autoloads; NakamaService can reference AuthManager directly.

```gdscript
# Phase 2: refresh window state on app resume + first load.
# Note: _notification (with underscore) is the Godot 4 virtual override for OS notifications.
func _notification(what: int) -> void:
    if what == NOTIFICATION_APPLICATION_RESUMED:
        # Re-subscribe to club topic (handles reinstall + token refresh edge cases — idempotent).
        if PlayerStore.club_id != "":
            NakamaService.subscribe_to_club_topic(PlayerStore.club_id)
        _refresh_window()

func _refresh_window() -> void:
    var window_resp = await NakamaService.get_current_window()
    if window_resp.get("ok", false):
        var data = window_resp.get("data", {})
        if typeof(data) == TYPE_DICTIONARY:
            var win = data.get("window", null)
            PlayerStore.current_window = win if typeof(win) == TYPE_DICTIONARY else {}
    _update_window_banner()

func _update_window_banner() -> void:
    var w: Dictionary = PlayerStore.current_window
    
    # Lower-division clubs never have API-Football data — always show "Coming soon".
    # club.division is available from PlayerStore if set; fall back to checking club_id suffix.
    var division: String = PlayerStore.get("club_division") if PlayerStore.get("club_division") != null else ""
    var is_lower_division := division in ["b_metro", "federal_a", "c_metro"]
    
    if w.is_empty():
        if is_lower_division:
            window_banner.text = "Coming soon — sin partidos vivos esta season"
            window_banner.modulate = Color(0.6, 0.6, 0.8, 1.0)  # muted blue — info
        else:
            window_banner.text = "Sin partidos próximos."
            window_banner.modulate = Color(0.5, 0.5, 0.5, 1.0)  # muted grey
        return
    
    var state: String = str(w.get("state", ""))
    match state:
        "scheduled":
            var secs: int = int(w.get("seconds_until_open", 0))
            var hrs: int = secs / 3600
            var mins: int = (secs % 3600) / 60
            window_banner.text = "Falta para que abra la ventana: %02d:%02d" % [hrs, mins]
            window_banner.modulate = Color(1.0, 0.8, 0.0, 1.0)  # gold — upcoming
        "open", "live":
            window_banner.text = "¡Ventana abierta! Tu club juega ahora."
            window_banner.modulate = Color(0.2, 0.8, 0.2, 1.0)  # green — active
        "closed", "cancelled":
            window_banner.text = "Ventana cerrada. Próximo partido próximamente."
            window_banner.modulate = Color(0.5, 0.5, 0.5, 1.0)  # grey — past
        _:
            window_banner.text = "Sin partidos próximos."
            window_banner.modulate = Color(0.5, 0.5, 0.5, 1.0)
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| FCM OS → FCMPlugin signal → NakamaService.register_fcm_token | Token value from Android OS to Nakama RPC — never logged, validated as string |
| server RPC → HomeScreen WindowBanner display | get_current_window response displayed directly; all fields from server (trusted source) |
| FCM OS → FCMPlugin.subscribeToTopic | Topic name built from AppConfig.FCM_TOPIC_PREFIX + PlayerStore.club_id — validated server-side by sendTopic before any FCM call |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-2-FCM-03 | Information Disclosure | register_fcm_token in NakamaService.gd | mitigate | Token passed to RPC but never logged on client side; NakamaService.gd only logs errors via push_warning |
| T-2-FCM-05 | Spoofing | subscribe_to_club_topic topic name | mitigate | Topic = FCM_TOPIC_PREFIX + club_id from PlayerStore (server-set); server-side validateTopicName enforced in sendTopic (plan 02-03); client cannot inject arbitrary topic |
| T-2-CACHE-01 | Tampering | HomeScreen window banner display | accept | Data comes from server-authoritative get_current_window RPC; client cannot modify; stale display for max one tick (~15 min) is acceptable |
| T-2-FCM-06 | Denial of Service | subscribeToTopic idempotency | mitigate | subscribed_topics.has(topic) check prevents repeated subscribe calls; PlayerStore.clear() resets on logout so re-subscribe on next login is clean |
</threat_model>
