// RPC: register_fcm_token
//
// Stores the device's FCM registration token for the authenticated user.
// D-10: In Phase 2, tokens are persisted but NOT used for per-user push sends.
//        Phase 4+ (Combate) will fire personal events (te atacaron, pibe preso) using these tokens.
// S-14: Singleton per userId — key='token' — new token overwrites old (one active device per account in Phase 2).
//        Phase 4 revisits with multi-device support (key='token_android' / 'token_ios').
//
// Input: { token: string, platform: "android" | "ios" }
// Output: { ok: true, registered: true } | { ok: false, error: string }

import { COL_FCM_TOKENS } from '../storage_keys';

export function rpcRegisterFcmToken(
  ctx: nkruntime.Context,
  logger: nkruntime.Logger,
  nk: nkruntime.Nakama,
  payload: string,
): string {
  const userId = ctx.userId;
  if (!userId) throw new Error('not_authenticated');

  let input: { token?: unknown; platform?: unknown } = {};
  try { input = JSON.parse(payload || '{}'); }
  catch (e) { return JSON.stringify({ ok: false, error: 'invalid_json_payload' }); }

  if (typeof input.token !== 'string' || input.token.length === 0)
    return JSON.stringify({ ok: false, error: 'token_required' });
  if (input.platform !== 'android' && input.platform !== 'ios')
    return JSON.stringify({ ok: false, error: 'invalid_platform' });

  nk.storageWrite([{
    collection: COL_FCM_TOKENS,
    key: 'token',
    userId,
    value: {
      platform: input.platform,
      registered_at: Date.now(),
      // Token NOT stored in plain value for log-safety — store in a separate sub-key.
      // Actually: store it (it's not a password — FCM tokens are client-generated, not secret in the
      // same way). But never LOG it (S4 pattern).
      token: input.token,
    },
    permissionRead: 0,
    permissionWrite: 0,
  }]);

  // S4: log userId + platform only — NEVER log the token value itself.
  logger.info('[register_fcm] user=%s platform=%s', userId, input.platform);
  return JSON.stringify({ ok: true, registered: true });
}

// RPC: get_current_window
//
// Returns the earliest upcoming (or currently open/live) match window for the
// authenticated player's club. Used by HomeScreen to display countdown / open banner.
//
// Input: {} (server reads ctx.userId → players/profile → club_id)
// Output: { ok: true, window: MatchWindow | null } | { ok: false, error: string }

import { COL_PLAYERS, COL_MATCH_WINDOWS, SYSTEM_USER_ID } from '../storage_keys';

export function rpcGetCurrentWindow(
  ctx: nkruntime.Context,
  logger: nkruntime.Logger,
  nk: nkruntime.Nakama,
  _payload: string,
): string {
  const userId = ctx.userId;
  if (!userId) throw new Error('not_authenticated');

  // Read player's profile to get club_id.
  const profileRead = nk.storageRead([{
    collection: COL_PLAYERS, key: 'profile', userId,
  }]);
  if (profileRead.length === 0) return JSON.stringify({ ok: false, error: 'no_profile' });
  const profile = profileRead[0].value as { club_id?: string };
  const clubId = profile.club_id;
  if (!clubId) return JSON.stringify({ ok: false, error: 'no_club' });

  // Scan match_windows for windows involving this club.
  // club_home_id and club_away_id use format 'team_{team_id}' from API-Football,
  // which won't match Phase 1 club UUIDs directly. In Phase 2, windows are auto-materialized
  // from API-Football data — the club IDs in windows are `team_{team_id}` strings.
  // The player's club_id (from Phase 1) is a UUID like 'boca_la_doce'.
  // For Phase 2, return the CLOSEST window regardless of club match (all players see heartbeat).
  // Phase 4+ will wire club_id mapping. Document this in SUMMARY.
  //
  // Note: This is an acceptable Phase 2 simplification — all players see the next window,
  // which demonstrates the heartbeat works. Accurate club filtering added in Phase 3/4
  // when team_id ↔ club_id mapping is built.

  const now = Date.now();
  const ACTIVE_STATES = ['scheduled', 'open', 'live'];

  let earliest: any = null;
  let cursor = '';
  for (let i = 0; i < 50; i++) {
    const page = nk.storageList(SYSTEM_USER_ID, COL_MATCH_WINDOWS, 100, cursor);
    for (const obj of (page.objects || [])) {
      const w = obj.value as any;
      if (!ACTIVE_STATES.includes(w.state)) continue;
      // Take the earliest opens_at that is either in the future or currently open.
      const relevantTime = w.state === 'scheduled' ? w.opens_at : now;
      if (earliest === null || relevantTime < (earliest.state === 'scheduled' ? earliest.opens_at : now)) {
        earliest = w;
      }
    }
    if (!page.cursor) break;
    cursor = page.cursor;
  }

  if (!earliest) return JSON.stringify({ ok: true, window: null });

  // Add computed helper field for client countdown.
  earliest.seconds_until_open = Math.max(0, Math.floor((earliest.opens_at - now) / 1000));
  logger.info('[get_window] user=%s club=%s state=%s', userId, clubId, earliest.state);
  return JSON.stringify({ ok: true, window: earliest });
}
