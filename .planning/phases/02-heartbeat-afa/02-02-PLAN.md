```typescript
function findNextKickoffWithin24h(nk: nkruntime.Nakama): number | undefined {
  const now = Date.now();
  const horizon = now + 24 * 3600 * 1000;
  let earliest: number | undefined;
  let cursor = '';
  for (let i = 0; i < 50; i++) {
    const page = nk.storageList(SYSTEM_USER_ID, COL_MATCH_WINDOWS, 100, cursor);
    for (const obj of (page.objects || [])) {
      const w = obj.value as { opens_at?: number; state?: string };
      if (w.state === 'scheduled' && w.opens_at !== undefined && w.opens_at > now && w.opens_at < horizon) {
        if (earliest === undefined || w.opens_at < earliest) earliest = w.opens_at;
      }
    }
    if (!page.cursor) break;
    cursor = page.cursor;
  }
  return earliest;
}
```

```typescript
// nakama/src/scheduler/seasons.ts
import { COL_FIXTURES, COL_META, SYSTEM_USER_ID, KEY_CURRENT_SEASON } from '../storage_keys';

interface SeasonState {
  season_id: number;
  division: 'primera' | 'nacional';
  torneo_name: string;
  started_at: number;
  ends_at: number;
  status: 'pre' | 'active' | 'ended';
}

export function detectSeasonState(
  _ctx: nkruntime.Context,
  logger: nkruntime.Logger,
  nk: nkruntime.Nakama,
): void {
  // Read all fixtures, find max season year + cluster last kickoff.
  const now = Date.now();
  let firstKickoff: number | undefined;
  let lastKickoff: number | undefined;
  let maxSeason = 0;

  let cursor = '';
  for (let i = 0; i < 50; i++) {
    const page = nk.storageList(SYSTEM_USER_ID, COL_FIXTURES, 100, cursor);
    for (const obj of (page.objects || [])) {
      const f = obj.value as { kickoff_utc?: number; season?: number; division?: string };
      if (f.division !== 'primera') continue; // D-19: only Primera drives season state
      if (typeof f.season === 'number' && f.season > maxSeason) maxSeason = f.season;
      if (typeof f.kickoff_utc === 'number') {
        if (firstKickoff === undefined || f.kickoff_utc < firstKickoff) firstKickoff = f.kickoff_utc;
        if (lastKickoff === undefined || f.kickoff_utc > lastKickoff) lastKickoff = f.kickoff_utc;
      }
    }
    if (!page.cursor) break;
    cursor = page.cursor;
  }

  if (!maxSeason || !firstKickoff || !lastKickoff) return; // no fixtures yet

  // D-17: auto-start when first fixture within <7d.
  // D-17: auto-end 7 days after last detected fixture of the season.
  const sevenDays = 7 * 24 * 3600 * 1000;
  let newStatus: SeasonState['status'] = 'pre';
  if (now >= lastKickoff + sevenDays) {
    newStatus = 'ended';
  } else if (firstKickoff - now <= sevenDays) {
    newStatus = 'active';
  }

  // Read existing season state.
  const r = nk.storageRead([{ collection: COL_META, key: KEY_CURRENT_SEASON, userId: SYSTEM_USER_ID }]);
  const existing: Partial<SeasonState> = r.length > 0 ? (r[0].value as SeasonState) : {};

  // Only write if status changed or not yet initialized.
  if (!r.length || existing.status !== newStatus || existing.season_id !== maxSeason) {
    const state: SeasonState = {
      season_id: maxSeason,
      division: 'primera',
      torneo_name: `Temporada ${maxSeason}`,
      started_at: existing.started_at ?? firstKickoff,
      ends_at: lastKickoff + sevenDays,
      status: newStatus,
    };
    nk.storageWrite([{
      collection: COL_META, key: KEY_CURRENT_SEASON, userId: SYSTEM_USER_ID,
      value: state, permissionRead: 0, permissionWrite: 0,
    }]);
    logger.info('[season] status=%s season=%d', newStatus, maxSeason);
  }
}
```

**Modification 1 — Optimistic concurrency in upsertFixture (MEDIUM 6 fix):**
Replace the simple `nk.storageWrite` with a read-modify-write pattern:

```typescript
function upsertFixture(nk: nkruntime.Nakama, logger: nkruntime.Logger, f: NormalizedFixture): void {
  // Optimistic concurrency: read version first to protect against admin_force_repoll ↔ tick race.
  // Rationale: tick_lock provides single-writer for scheduled ticks; 'version' catches the
  // narrow race window where admin_force_repoll fires while a tick is committed but not yet locked.
  const existing = nk.storageRead([{
    collection: COL_FIXTURES, key: f.fixture_id, userId: SYSTEM_USER_ID,
  }]);
  const version = existing.length > 0 ? existing[0].version : undefined;
  try {
    nk.storageWrite([{
      collection: COL_FIXTURES, key: f.fixture_id, userId: SYSTEM_USER_ID,
      value: f,
      ...(version !== undefined ? { version } : {}),
      permissionRead: 2, permissionWrite: 0,
    }]);
  } catch (e) {
    const msg = String(e);
    if (msg.includes('version_mismatch') || msg.includes('version conflict')) {
      // Retry once without version (take latest).
      logger.warn('[api_football] version_mismatch on fixture=%s — retrying without version', f.fixture_id);
      try {
        nk.storageWrite([{
          collection: COL_FIXTURES, key: f.fixture_id, userId: SYSTEM_USER_ID,
          value: f, permissionRead: 2, permissionWrite: 0,
        }]);
      } catch (e2) {
        logger.error('[api_football] retry also failed for fixture=%s: %s', f.fixture_id, String(e2));
        // Skip — next tick will try again.
      }
    } else {
      throw e; // Unexpected error — propagate.
    }
  }
}
```

```typescript
// buildClubTeamMap: Fetches all teams for both AFA leagues and builds a mapping
// from Phase 1 club_id (lunfardo slug) to API-Football team_id.
//
// Called ONCE per season (or when admin_force_repoll is called with rebuild=true in future).
// Writes to meta:club_team_map and meta:unmatched_clubs.
// Fuzzy match: normalize team name + city to lowercase, strip accents, match against club
// lunfardo_name + barrio_hq fields from Phase 1 clubs collection.
//
// Phase 1 clubs are in COL_CLUBS (SYSTEM_USER_ID). Each club has:
//   { id, lunfardo_name, barrio_hq, division, ... }
//
// Match heuristic: if api_team_name_normalized contains any word from lunfardo_name_normalized,
// consider it a candidate. Manual override available via admin_set_club_team_mapping RPC (plan 02-05).

export function buildClubTeamMap(
  ctx: nkruntime.Context,
  logger: nkruntime.Logger,
  nk: nkruntime.Nakama,
): void {
  const leagueIdsRead = nk.storageRead([{
    collection: COL_META, key: KEY_API_FOOTBALL_LEAGUE_IDS, userId: SYSTEM_USER_ID,
  }]);
  if (leagueIdsRead.length === 0) {
    logger.warn('[club_map] no league IDs stored yet — run pollFixtures first');
    return;
  }
  const leagueIds = leagueIdsRead[0].value as { primera_id?: number; nacional_id?: number };
  const year = new Date().getUTCFullYear();
  const apiKey = ctx.env['API_FOOTBALL_KEY'];
  if (!apiKey) {
    logger.warn('[club_map] API_FOOTBALL_KEY missing — skipping club_team_map build');
    return;
  }

  // Read Phase 1 clubs.
  const clubs: Array<{ id: string; lunfardo_name: string; barrio_hq?: string }> = [];
  let cursor = '';
  for (let i = 0; i < 20; i++) {
    const page = nk.storageList(SYSTEM_USER_ID, 'clubs', 100, cursor);
    for (const obj of (page.objects || [])) {
      const c = obj.value as { id?: string; lunfardo_name?: string; barrio_hq?: string };
      if (c.id && c.lunfardo_name) clubs.push({ id: c.id, lunfardo_name: c.lunfardo_name, barrio_hq: c.barrio_hq });
    }
    if (!page.cursor) break;
    cursor = page.cursor;
  }

  const normalize = (s: string): string =>
    s.toLowerCase().normalize('NFD').replace(/[̀-ͯ]/g, '').replace(/[^a-z0-9 ]/g, ' ').trim();

  const clubTeamMap: Record<string, number> = {};
  const unmatchedClubs: Record<string, { league_id: number; team_id: number; city: string }> = {};

  for (const leagueId of [leagueIds.primera_id, leagueIds.nacional_id].filter(Boolean) as number[]) {
    let teamsResp: any;
    try {
      const r = nk.httpRequest(
        `https://v3.football.api-sports.io/teams?league=${leagueId}&season=${year}`,
        'get',
        { 'x-apisports-key': apiKey },
        undefined,
        8000,
      );
      if (r.code !== 200) {
        logger.warn('[club_map] /teams returned %d for league=%d', r.code, leagueId);
        continue;
      }
      teamsResp = JSON.parse(r.body);
    } catch (e) {
      logger.error('[club_map] /teams fetch failed for league=%d: %s', leagueId, String(e));
      continue;
    }

    for (const entry of (teamsResp.response || [])) {
      const apiTeam = entry.team as { id?: number; name?: string; city?: string };
      if (!apiTeam.id || !apiTeam.name) continue;
      const apiNameNorm = normalize(apiTeam.name);
      const apiWords = new Set(apiNameNorm.split(' ').filter(w => w.length > 2));

      // Try to match against Phase 1 clubs.
      let bestMatch: string | null = null;
      let bestScore = 0;
      for (const club of clubs) {
        const clubNorm = normalize(club.lunfardo_name);
        const clubWords = clubNorm.split(' ').filter(w => w.length > 2);
        const overlap = clubWords.filter(w => apiWords.has(w)).length;
        if (overlap > bestScore) {
          bestScore = overlap;
          bestMatch = club.id;
        }
      }

      if (bestMatch && bestScore >= 1) {
        clubTeamMap[bestMatch] = apiTeam.id;
        logger.info('[club_map] matched club_id=%s → team_id=%d (%s)', bestMatch, apiTeam.id, apiTeam.name);
      } else {
        unmatchedClubs[apiTeam.name] = {
          league_id: leagueId,

    collection: COL_META, key: 'unmatched_clubs', userId: SYSTEM_USER_ID,
    value: unmatchedClubs, permissionRead: 0, permissionWrite: 0,
  }]);
  logger.info('[club_map] built: %d matched, %d unmatched', Object.keys(clubTeamMap).length, Object.keys(unmatchedClubs).length);
}
```

```typescript
  nk: nkruntime.Nakama,
  homeTeamId: number,

  const mapRead = nk.storageRead([{ collection: COL_META, key: 'club_team_map', userId: SYSTEM_USER_ID }]);

    club_away_id: reverseMap[awayTeamId] ?? null,
  };
}
```
