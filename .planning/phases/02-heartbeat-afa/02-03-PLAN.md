```typescript
function base64ToUtf8(b64: string, nk: nkruntime.Nakama): string {
  const ab = nk.base64Decode(b64);
  const bytes = new Uint8Array(ab as any);
  let s = '';
  for (let i = 0; i < bytes.length; i++) s += String.fromCharCode(bytes[i]);
  return s;
}
```

```typescript
// util/topic_name.ts
// Validates FCM topic names per FCM v1 API spec (RESEARCH.md §S-5, §Q3).
// Topic names must match [a-zA-Z0-9_.~%-]+ — letters, digits, underscore, dot, tilde, percent.
// Phase 1 clubs use lowercase + underscores exclusively — all valid.
// Length cap: 900 chars (conservative; FCM has no official max but widely tested at 100).

export interface TopicNameResult {
  ok: boolean;
  sanitized?: string;   // the validated topic name, only when ok: true
  error?: string;
}

const TOPIC_MAX_LENGTH = 900;
const TOPIC_ALLOWED_RE = /^[a-zA-Z0-9_.~%-]+$/;

export function validateTopicName(raw: unknown): TopicNameResult {
  if (typeof raw !== 'string') {
    return { ok: false, error: 'topic_must_be_string' };
  }
  const topic = raw.trim();
  if (topic.length === 0) {
    return { ok: false, error: 'topic_empty' };
  }
  if (topic.length > TOPIC_MAX_LENGTH) {
    return { ok: false, error: 'topic_too_long' };
  }
  if (!TOPIC_ALLOWED_RE.test(topic)) {
    return { ok: false, error: 'invalid_topic_chars' };
  }
  return { ok: true, sanitized: topic };
}
```

```typescript
// RPC: admin_test_validate_topic

// Test-harness RPC that exposes validateTopicName for deterministic test assertions.
// Gated by ADMIN_TEST_MODE=true — NOT available in production.
//
// Input: { topic_in: string }
// Output: { ok: bool, normalized?: string, error?: string }

import { validateTopicName } from '../util/topic_name';
import { requireAdmin } from '../util/admin_auth';

export function rpcAdminTestValidateTopic(
  ctx: nkruntime.Context,
  logger: nkruntime.Logger,
  nk: nkruntime.Nakama,
  payload: string,
): string {
  if (ctx.env['ADMIN_TEST_MODE'] !== 'true') {
    return JSON.stringify({ ok: false, error: 'test_mode_disabled' });
  }
  const auth = requireAdmin(ctx, logger);
  if (!auth.ok) return JSON.stringify({ ok: false, error: auth.error });

  let input: { topic_in?: unknown } = {};
  try { input = JSON.parse(payload || '{}'); }
  catch (e) { return JSON.stringify({ ok: false, error: 'invalid_payload' }); }

  const result = validateTopicName(input.topic_in);
  logger.info('[admin][test] validate_topic topic_in=%s ok=%s', String(input.topic_in), result.ok);
  return JSON.stringify({
    ok: result.ok,
    ...(result.ok ? { normalized: result.sanitized } : { error: result.error }),
  });
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Nakama server → Google OAuth2 | Service account private key (PEM) travels in memory only; loaded from env var, never logged |
| Nakama server → FCM API | OAuth2 access token in Authorization header; token cached server-side; never sent to client |
| FCM API → Android devices | FCM delivers push; Nakama has no visibility into delivery confirmation |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-2-FCM-01 | Spoofing | sendTopic topic param | mitigate | validateTopicName called before any FCM HTTP call; invalid chars rejected with log (VALIDATION.md 02-04-DAY03-topic) |
| T-2-FCM-02 | Denial of Service | sendTopic per window | mitigate | notified_open_at set in same storageWrite that transitions state (anti-double-push); even if tick fires twice, second call skips send (VALIDATION.md 02-04-DAY03-once) |
| T-2-FCM-03 | Information Disclosure | FCM_SERVICE_ACCOUNT_B64 | mitigate | Never logged — loadServiceAccount reads it, uses it for JWT, discards; private_key field is PEM, not UUIDs that look like tokens |
| T-2-FCM-04 | Spoofing | OAuth2 token cache | accept | Cached in COL_META (server-only, permissionRead:0); client cannot read; worst case stale token causes FCM 401 → logged, next tick refreshes |
| T-2-INJ-01 | Injection | FCM topic name from club_id | mitigate | validateTopicName regex [a-zA-Z0-9_.~%-]+ enforced; Phase 1 clubs already use safe format; WARN+skip on violation |
</threat_model>
