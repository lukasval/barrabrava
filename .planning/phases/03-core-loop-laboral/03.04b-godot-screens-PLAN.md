## HomeScreen.gd extension (per UI-SPEC §5.1)
```gdscript
# Add new @onready vars for ResourceRow widgets + TurnoButton + QuickActions row.
@onready var resource_row: HBoxContainer = $TopArea/ResourceRow
@onready var plata_widget: Control = $TopArea/ResourceRow/PlataWidget
@onready var aguante_widget: Control = $TopArea/ResourceRow/AguanteWidget
@onready var rep_widget: Control = $TopArea/ResourceRow/ReputacionWidget
@onready var vbc_widget: Control = $TopArea/ResourceRow/VBCWidget
@onready var turno_button: Button = $Content/TurnoButton
@onready var laburar_button: Button = $Content/QuickActions/Laburar
@onready var reclutar_button: Button = $Content/QuickActions/Reclutar
@onready var aguantadero_button: Button = $Content/QuickActions/Aguantadero
@onready var idle_notice: Label = $Content/IdleNotice
@onready var rank_badge: Control = $TopBar/RankBadge

func _ready() -> void:
  # Keep existing Phase 1/2 _ready behaviour.
  # ADD:
  PlayerStore.resources_updated.connect(_refresh_resource_widgets)
  PlayerStore.roster_updated.connect(_refresh_turno_button)
  turno_button.pressed.connect(_open_turno_modal)
  laburar_button.pressed.connect(func(): FlowRouter.go_roster())
  reclutar_button.pressed.connect(func(): FlowRouter.go_recruit())
  aguantadero_button.pressed.connect(func(): FlowRouter.go_aguantadero())
  await PlayerStore.refresh_resources_and_roster()
  _refresh_resource_widgets()
  _refresh_turno_button()

func _notification(what: int) -> void:
  if what == NOTIFICATION_APPLICATION_RESUMED:
    await PlayerStore.refresh_resources_and_roster()
    _refresh_resource_widgets()
    _refresh_turno_button()
    # Keep Phase 2 window refresh.
    _refresh_window()

func _refresh_resource_widgets() -> void:
  plata_widget.set_value(PlayerStore.plata)
  aguante_widget.set_value(0)  # aguante grupal — fetch from barra_state next milestone
  rep_widget.set_value(PlayerStore.reputacion)
  vbc_widget.set_value(PlayerStore.vbc)
  rank_badge.set_rank(PlayerStore.rank)

func _refresh_turno_button() -> void:
  var win = PlayerStore.current_window
  var visible = typeof(win) == TYPE_DICTIONARY and (
    win.get("state", "") == "open" or win.get("state", "") == "live"
  )
  turno_button.visible = visible
  if visible:
    var eligible = _count_eligible_pibes()
    turno_button.disabled = eligible == 0
    turno_button.text = "Hacer turno" if eligible > 0 else "Sin pibes con energía"

func _count_eligible_pibes() -> int:
  var c = 0
  for p in PlayerStore.pibes:
    if typeof(p) == TYPE_DICTIONARY and int(p.get("energia", 0)) >= 30:
      var et = p.get("en_turno_until", null)
      if et == null or int(et) < int(Time.get_unix_time_from_system() * 1000):
        c += 1
  return c

func _open_turno_modal() -> void:
  var modal_scene = preload("res://scenes/TurnoModal.tscn")
  var modal = modal_scene.instantiate()
  add_child(modal)
  modal.show_modal(PlayerStore.current_window)
```

## TutorialScreen.gd EXTENSION — 6-step state machine + elapsed_ms telemetry (UI-SPEC §10)
```gdscript
extends Control

@onready var step_indicator: Label = $TopBar/StepIndicator
@onready var skip_button: Button = $TopBar/Skip
@onready var step_title: Label = $Content/Title
@onready var step_body: RichTextLabel = $Content/Body
@onready var step_illustration: Control = $Content/Illustration
@onready var dots: HBoxContainer = $Content/Dots
@onready var back_button: Button = $Footer/Back
@onready var cta_button: Button = $Footer/CTA

const STEPS: Array = [
  { "id": "welcome",          "title": "Bienvenido a la barra",   "body": "Tu pibé llegó al barrio. Te vamos a mostrar el laburo en 5 pasos.", "cta": "Dale, vamos" },
  { "id": "recruit_intro",    "title": "Sumá tu primer pibé",     "body": "Cada día llegan 3 pibes nuevos al barrio. Elegí uno para sumarse a tu barra.", "cta": "Ir a reclutar" },
  { "id": "assign_profession","title": "Ponelo a laburar",        "body": "Tu pibé puede laburar de trapito, vendedor, patovica o remisero. Genera Plata mientras vos no estás.", "cta": "Asignar profesión" },
  { "id": "see_plata",        "title": "Cobrá la Plata",          "body": "Cada hora que tu pibé labura, suma Plata al tope. Cobrala cuando vuelvas. Cap: 12 horas.", "cta": "Entendido" },
  { "id": "first_turno",      "title": "Hacé tu primer turno",    "body": "Cuando tu club juega, abrí la ventana y mandá a tus pibes al estadio. Eso le suma Aguante al pozo grupal.", "cta": "Hacer turno (simulado)" },
  { "id": "reward",           "title": "Bienvenida con regalo",   "body": "Para arrancar bien, te damos tu primer trapo y un cántico. Aplaudila al equipo.", "cta": "Empezar de verdad" },
]

var _step: int = 1
var tutorial_start_at_ms: int = 0  # captured on step 1 first render; used for elapsed_ms telemetry

func _ready() -> void:
  # CRITICAL: capture start timestamp on first render so elapsed_ms reflects
  # the full tutorial duration (LAB-TUTORIAL-DURATION invariant in plan 03.05).
  tutorial_start_at_ms = int(Time.get_unix_time_from_system() * 1000)
  skip_button.pressed.connect(_on_skip)
  back_button.pressed.connect(_on_back)
  cta_button.pressed.connect(_on_cta)
  _render_step()

func _elapsed_ms() -> int:
  return int(Time.get_unix_time_from_system() * 1000) - tutorial_start_at_ms

func _render_step() -> void:
  var info = STEPS[_step - 1]
  step_indicator.text = "Paso %d de %d" % [_step, STEPS.size()]
  step_title.text = info["title"]
  step_body.text = info["body"]
  cta_button.text = info["cta"]
  back_button.visible = (_step > 1)
  _render_dots()

func _render_dots() -> void:
  for i in range(dots.get_child_count()):
    var dot: ColorRect = dots.get_child(i)
    dot.color = AppTheme.ACCENT if i + 1 == _step else AppTheme.BORDER_INACTIVE

func _on_cta() -> void:
  cta_button.disabled = true
  match _step:
    1:
      _step = 2
      _render_step()
    2:
      # v1: tap-through acknowledgement. Step 2 is a mock-up of RecruitScreen;
      # the real first recruit happens post-tutorial via HomeScreen guided hint.
      await FlowRouter.tutorial_advance(2, _elapsed_ms())
      _step = 3
      _render_step()
    3:
      await FlowRouter.tutorial_advance(3, _elapsed_ms())
      _step = 4
      _render_step()
    4:
      # Step 4: acknowledgement of the idle-collect mechanic. Granted starter
      # +10 Plata via collect_idle tutorial bypass (see plan 03.03 collect_idle).
      await NakamaService.collect_idle()
      await FlowRouter.tutorial_advance(4, _elapsed_ms())
      _step = 5
      _render_step()
    5:
      # v1: simulated turno acknowledgement — server credits +20 Rep via the
      # tutorial-bypass path in complete_tutorial (plan 03.03). The actual first
      # turno happens post-tutorial via HomeScreen TurnoButton when a real window opens.
      await FlowRouter.tutorial_advance(5, _elapsed_ms())
      _step = 6
      _render_step()
    6:
      # FINAL — grants trapo + cántico atomically. Server logs tutorial_duration_ms
      # at this step for LAB-TUTORIAL-DURATION invariant (plan 03.05).
      await FlowRouter.tutorial_advance(6, _elapsed_ms())
      # FlowRouter routes home when tutorial_done.
  cta_button.disabled = false

func _on_back() -> void:
  if _step > 1:
    _step -= 1
    _render_step()

func _on_skip() -> void:
  var dlg = ConfirmationDialog.new()
  dlg.dialog_text = "¿Saltar el tutorial? Vas a perderte la recompensa de bienvenida."
  dlg.ok_button_text = "Sí, saltar"
  dlg.cancel_button_text = "Seguir"
  add_child(dlg)
  dlg.confirmed.connect(func():
    # Mark tutorial done server-side via final step. elapsed_ms still captures
    # the (short) time the user actually spent before skipping.
    await FlowRouter.tutorial_advance(6, _elapsed_ms())
  )
  dlg.popup_centered()
```
